    def method_missing(name, *a, &b)
      name = name.to_s
      case name
      when /\=$/
        self[name.chomp('=')] = a.first
      when /\?$/
        self[name.chomp('?')]
      when /\!$/
        super
      else
        self[name]
      end
    end

# File lib/path/store.rb, line 129
    def prime!(hash=nil)
      if hash
        hash.each do |k,v|
          case v
          when Hash
            self[k] = self.class.new(File.join(@dir,k),v)
          when self.class
            self[k] = self.class.new(File.join(@dir,k),v.to_h)
          else
            self[k] = v
          end
        end
      else
        entries = Dir.entries(@dir) - [ '.', '..' ]
        entries.each do |fname|
          v = self[fname]
          v.prime! if self.class === v
        end
      end
    end

    def read!(name)
      file = File.join(@dir, name)
      if File.file?(file)
        YAML.load(File.new(file))
      elsif File.directory?(file)
        if @depth.zero?
          self.class.new(file)
        else
          self.class.new(file, depth - 1)
        end
      else
        if @depth.zero?
          nil
        else
          self.class.new(file)
        end
      end
    end

    def save!
      @data.each do |name, value|
        file  = File.join(@dir,name)
        case value
        when self.class
          FileUtils.mkdir_p(@dir)
          value.save!
        else
          old = File.read(file) if File.exist?(file)
          new = value.to_yaml
          if old != new
            FileUtils.mkdir_p(@dir)
            File.open(file, 'w'){ |f| f << new }
          end
        end
      end
    end
